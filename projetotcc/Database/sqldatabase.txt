CREATE TABLE funcionario (
    id Serial NOT NULL,
    id_funcionario INT NOT NULL,
    nome VARCHAR(255),
    CONSTRAINT pk_func PRIMARY KEY (id)
);

CREATE TABLE operacao (
    acao VARCHAR(7) NOT NULL,
    CONSTRAINT pk_acao PRIMARY KEY (acao)
);

CREATE TABLE registro (
    id_registro Serial NOT NULL,
    hora TIME,
    data DATE,
    id_funcionario BIGINT,
    acao VARCHAR(7),
    CONSTRAINT pk_registro PRIMARY KEY (id_registro),
    CONSTRAINT fk_registro_func FOREIGN KEY (id_funcionario) REFERENCES funcionario(id),
    CONSTRAINT fk_registro_acao FOREIGN KEY (acao) REFERENCES operacao(acao)
);
